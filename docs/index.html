Historical High NotES

  • Netscape and Sun employ Brendan Eich to create a scripting language to interface with Java (1995)
  • In about 2 weeks, Eich created mocha
  • ™ issues with "mocha" ― re-branded as LiveScript
  • Popularity of Java ― re-branded as JavaScript
  • Netscape asked Ecma to maintain ISO standard (1996)
  • Oracle owned JavaScript™ ― maintained as ECMAScript (1997)
  • Specification, ECMA-262, maintained by TC39

History of ReleasES

  • 1997 ― ES1 published
  • 1998 ― ES2 published
  • 2002 ― ES3 published
  • JavaScript (ECMAScript) gets famous
  • ES4 skipped because of reasons
  • 2011 ― ES5 published
  • 2015 ― ES6 published
  • TC39 transitions to yearly release schedule
  • 2016 ― ES7 (ES2016) published
  • 2017 ― ES8 (ES2017) published in June

ES6 const and let

WidESpread Usage

  • const ― use by default
  • let ― use when re-assignment is required
  • var ― "old 'n busted'"

const

Block scoped, cannot be re-assigned or redeclared


    const powerLevel = 8999;
    powerLevel = powerLevel + 2;// TypeError!
        

    const powerLevel;// SyntaxError!
        

    function foo() {
        const x = 1;
        if (true) {
            const x = 2;// x === 2
        }
        // x === 1
    }

let

Block scoped and re-assignable


    let powerLevel = 8999;
    powerLevel = powerLevel + 2;// It's over 9000!
        

    if (/* condition */) {
        let foo;
        let foo;// SyntaxError!
    }
        

    function foo() {
        let x = 1;
        if (true) {
            let x = 2;// x === 2
        }
        // x === 1
    }

The DEScent of var

Variables declared with var DO NOT have block scope


    var x = 1;
    {
      var x = 2;
    }
    console.log(x);// 2
        

    let x = 1;
    {
      let x = 2;
    }
    console.log(x);// 1

Things to remember

let bindings are not subject to Variable Hoisting


    function someFunction() {
        console.log(bar);// undefined
        console.log(foo);// ReferenceError
        var bar = 1;
        let foo = 2;
    }

        'prefer-const': ['error']// autofix
    }

ES6 Arrow Functions

ES6 Arrow Functions

Language comparison


    -- Haskell
    addOne = \x -> x + 1
                

    -- Elm
    addOne = \x -> x + 1
                

    /* ReasonML */
    let addOne = (x) => x + 1;
                

    // JavaScript
    const addOne = x => x + 1;

Arrow Functions

No arguments (0-arity)


    // ES5
    var greet = function() {
        alert('Good Morning');
    };
            

    // ES6
    const greet = () => alert('Good Morning');

Arrow Functions

One argument (1-arity)


    // ES5
    var isEven = function(num) {
        return num % 2 === 0;
    };
            

    // ES6
    const isEven = num => (num % 2 === 0);
            

    // Default values
    const foo = (val = 42) => console.log(val);
    foo(9001);// 9001
    foo();// 42

Arrow Functions

Multiple arguments (n-arity, n > 1)


    // ES5
    var add = function(a, b) {
        return a + b;
    };
            

    // ES6
    const add = (a, b) => a + b;

Arrow Functions

Higher order functions (ex: currying)


    // ES5
    var add = function(a) {
        return function(b) {
            return a + b;
        };
    };
            

    // ES6
    const add = a => b => a + b;
            

    // usage
    const addFive = add(5);
    add(1)(2) === 3;// true
    addFive(6) === 11;// true

Arrow Functions in Action


    // NO-OP
    const noop = () => {};
            

    // Identity
    const identity = val => val;
            

    // "immutable" arrays
    const concat = (a, b) => [...a, ...b]
            

    // Multi-line functions
    const isValid = val => {
        const MIN = 42;
        const MAX = 9000;
        return (val >= MIN) && (val <= MAX);
    };

Things to remember

Non-binding of this


    let Brain = function() {this.thought = 'I love ES';};
        

    // YEP
    Brain.prototype.think = function() {
        console.log(this.thought);
    };
    const myBrain = new Brain();
    myBrain.think();// 'I love ES'
            

    // NOPE
    Brain.prototype.think = () => {
        console.log(this.thought);
    };
    const myBrain = new Brain();
    myBrain.think();// this is not defined

Things to remember

Non-binding of arguments


    // YEP
    const repeat = function() {
        console.log(Array.prototype.slice.call(arguments));
    };
    repeat('foo', 'bar');// ['foo', 'bar']
            

    // NOPE
    const repeat = () => {
        console.log(Array.prototype.slice.call(arguments));
    };
    repeat('foo', 'bar');// arguments is undefined

Things to remember

Add (...) when returning objects directly


    // YEP
    const getOptions = () => ({
        color: 'blue',
        size: 'small'
    });
    getOptions();// {color: 'blue', size: 'small'}
            

    // NOPE
    const getOptions = () => {
        color: 'blue',
        size: 'small'
    };
    getOptions();// undefined

Template Literals

Tagged template literals


    // No ES5 analog
            

    const button = text => ``;

    button`push me`// ""

        const truth = 'JavaScript';
        language = (language !== truth) ? truth : language;
        return text[0] + language;
    };
    const thing = 'Java';
    fix`I love ${thing}`// "I love JavaScript"

ES6 Destructuring Assignment

DEStructuring Assignment

Objects


    const data = {height: 42, width: 43};
            

    // Assignment
    const {height, width} = data;
    console.log(height);// 42
            

    // Assignment with new names
    const {height: h, width: w} = data;
    console.log(h);// 42
            

    // Default values
    const {height, width, depth = 9001} = data;
    console.log(depth);// 9001

DEStructuring Assignment

Objects ― Advanced usage


    const data = {height: 42, width: 43};
            

    // Computed property assignment
    const key = 'height';
    const {[key]: h} = data;
    console.log(height);// ReferenceError!
    console.log(h);// 42
            

    // Assignment separate from declaration
    let foo;
    ({foo} = {foo: 42})
    console.log(foo);// 42

DEStructuring Assignment

Arrays ― Similar to object destructuring


    const numbers = [1, 2, 3, 4, 5];
            

    // Assignment
    const [first, second] = numbers;
    console.log(first, second);// 1 2
            

    // Variable swap (re-assignment)
    let a = 1;
    let b = 2;
    [b, a] = [a, b];
    console.log(a, b);// 2 1

DEStructuring Assignment

Arrays ― Advanced usage


    const numbers = [1, 2, 3, 4, 5];
            

    // Ignore values
    const [first, , third, , fifth] = numbers;
    console.log(first, third, fifth);// 1 3 5
            

    // Default values
    const [one, two, three, four, five, six = 6] = numbers;
    console.log(six);// 6
            

    // Obviate magic numbers - 0, 1, 2, 3
    const first = numbers[0];// with magic number
    const [first] = numbers;// without magic number

ES6 Rest and Spread

RESt parameters

From arguments to an array


    // ES5
    function f(a, b) {
      var args = Array.prototype.slice.call(arguments, f.length);
      // do something with args array
    }
            

    // ES6
    function f(a, b, ...args) {
      // do something with args array
    }

Spread Syntax

Say goodbye to apply


    function f(x, y, z) {}
    const args = [1, 2, 3];
            

    // ES5
    f.apply(null, args);
            

    // ES6
    f(...args);

Spread Syntax

Using lodash to spread function arguments


    function f(a, b, c) {}
    const args = [1, 2, 3];
            

    const _ = require('lodash');

    // _.spread is based on the spread operator
    _.spread(f)(args);

Spread syntax

Array operations


    // array clone (shallow)
    let foo = [1, 2, 3];
    let bar = [...foo];
    foo[2] = 'boot';
    console.log(foo);// [1, 2, "boot"]
    console.log(bar);// [1, 2, 3]
            

    // array concatenation
    const dig = [1, 2, 3];
    const its = [4, 5, 6];
    const digits = [...dig, ...its];// [1, 2, 3, 4, 5, 6]

Things to remember

Rest/Spread syntax for object properties is still Stage 3


    // Rest properties
    const {a, b, ...other} = {a: 1, b: 2, c: 3, d: 4};
    a;// 1
    b;// 2
    other;// {c: 3, d: 4}
            

    // Spread properties
    const n = {a, b, ...other};
    n;// {a: 1, b: 2, c: 3, d: 4}

ESLint Rules for Rest & Spread Syntax


    rules: {
        'prefer-rest-params': ['warn'],
        'prefer-spread': ['error'],// autofix
        'rest-spread-spacing': ['error']// autofix
    }
  • prefer-rest-params
  • refer-spread
  • rest-spread-spacing

Set

lets you store unique values of any type


    // items are unique
    let ids = new Set([1, 1, 1, 2, 3, 3]);// Set {1, 2, 3}
        

    // familiar API
    ids.add(4);// Set {1, 2, 3, 4}
    ids.delete(1);// Set {2, 3, 4}
    ids.has(3);// true
    ids.size;// 3
    ids.clear();// ids.size === 0
        

    // iterate through elements in insertion order
    ids.add(1).add(2).add(3);// Set {1, 2, 3}
    ids.forEach(id => console.log(id));

ES8 Async functions

Overview

  • Part of ES8 (ES2017) specification
  • Supported in FF 52+, Chrome 55+ and Node 8+
  • await can only be used inside an async function

Let's get asynchronous


    async function getAllTheThings() {
        const foo = await getStuff();// async operation
        const bar = await getMoreStuff();// also async
        console.log('All done!');
        return [...foo, ...bar];
    }
        

    getAllTheThings();// ... "All done!"
        

    // you can also use Promise.prototype.then
    getAllTheThings().then(data => {
        data.forEach(item => console.log(item));
    });

Example Using jQuery


    const $ = require('jquery');
    const url = '/some/api/endpoint';
        

    // ES5
    $.get({url: url}).then(function() {
        console.log('All done!');
    });
        

    // ES8
    (async () => {
        await $.get({url});
        console.log('All done!');
    })();
